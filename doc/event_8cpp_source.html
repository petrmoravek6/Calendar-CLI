397 CEvent CEvent::makeEventWithNewNearestDate(const STimeStamp & lowLimit, const set<STimePeriod> & intervals) const {
398  STimeStamp lastCoveredPoint = lowLimit;
399  for (const auto & interval : intervals) {
400  // we have a gap
401  if (interval.m_Start > lastCoveredPoint) {
402  // we have a gap big enough to put the event there

404  if (lastCoveredPoint.minutesBetween(interval.m_Start) >= durationOfThisEvent) {

411  m_Participants, m_Details, newRepeat, m_IsMovable};
